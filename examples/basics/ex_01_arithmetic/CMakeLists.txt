# ==================================================================
#  tubex-lib / example - cmake configuration file
# ==================================================================

cmake_minimum_required (VERSION 3.0.2)

project (TUBEX_BASICS_01 LANGUAGES CXX)

include (FindPkgConfig)
pkg_search_module (TUBEX REQUIRED tubex)
message (STATUS "Found Tubex version ${TUBEX_VERSION}")

set(CMAKE_CXX_STANDARD 11)
add_compile_options(-O3 -DNDEBUG)

################################################################################
# With Gaol, we add "-Wno-deprecated" to remove lots of warnings generated by
# a function definition in gaol_interval.h
include(CheckCXXCompilerFlag)
CHECK_CXX_COMPILER_FLAG ("-Wno-deprecated" COMPILER_SUPPORTS_WNO_DEPRECATED)
if (COMPILER_SUPPORTS_WNO_DEPRECATED)
  set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wno-deprecated")
endif()
################################################################################

if (WIN32)
  add_definitions(-U__STRICT_ANSI__)
endif ()



add_executable(basics_01_arithmetic main.cpp)

target_compile_options (basics_01_arithmetic PUBLIC ${TUBEX_CFLAGS})
target_include_directories (basics_01_arithmetic PUBLIC ${TUBEX_INCLUDE_DIRS})
target_link_libraries (basics_01_arithmetic PUBLIC ${TUBEX_LDFLAGS})
